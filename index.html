<script>

const WEATHERAPI_KEY = "YOUR_API_KEY";

let currentSource = "openmeteo";

let forecastData = [];

let coords = null;

const themes = {

    Clear:{

        background:"linear-gradient(135deg,#f59e0b,#fb923c,#0ea5e9)",

        icon:'<i class="fa-solid fa-sun text-yellow-200"></i>',

        title:"Ясно",

        effect:"sun"

    },

    Rain:{

        background:"linear-gradient(135deg,#0f172a,#1e3a8a,#312e81)",

        icon:'<i class="fa-solid fa-cloud-rain text-sky-200"></i>',

        title:"Дождь",

        effect:"rain"

    },

    Snow:{

        background:"linear-gradient(135deg,#64748b,#94a3b8,#cbd5e1)",

        icon:'<i class="fa-solid fa-snowflake text-white"></i>',

        title:"Снег",

        effect:"snow"

    },

    Clouds:{

        background:"linear-gradient(135deg,#334155,#475569,#64748b)",

        icon:'<i class="fa-solid fa-cloud text-slate-100"></i>',

        title:"Облачно",

        effect:"clouds"

    }

};

function applyTheme(type){

    const t = themes[type] || themes.Clouds;

    document.body.style.background = t.background;

    document.getElementById("main-icon").innerHTML = t.icon;

    document.getElementById("weather-badge").innerText = t.title;

    renderEffects(t.effect);

}

function renderEffects(effect){

    const c = document.getElementById("effects");

    c.innerHTML = "";

    if(effect==="sun"){

        const s=document.createElement("div");

        s.style.cssText="position:absolute;top:-140px;right:-140px;width:480px;height:480px;border-radius:50%;background:radial-gradient(circle,rgba(255,220,120,0.55),transparent 70%);animation:pulse 6s infinite;";

        c.appendChild(s);

    }

    if(effect==="rain"){

        for(let i=0;i<80;i++){

            const d=document.createElement("div");

            d.className="rain";

            d.style.left=Math.random()*100+"%";

            d.style.animationDuration=(0.5+Math.random())+"s";

            c.appendChild(d);

        }

    }

    if(effect==="snow"){

        for(let i=0;i<50;i++){

            const s=document.createElement("div");

            s.className="snow";

            s.style.left=Math.random()*100+"%";

            s.style.animationDuration=(4+Math.random()*4)+"s";

            c.appendChild(s);

        }

    }

    if(effect==="clouds"){

        for(let i=0;i<6;i++){

            const cl=document.createElement("div");

            cl.className="cloud";

            cl.style.top=i*80+"px";

            cl.style.width=(180+Math.random()*120)+"px";

            cl.style.height="60px";

            c.appendChild(cl);

        }

    }

}

function selectDay(i){

    const d=forecastData[i];

    document.querySelectorAll(".forecast-item").forEach(x=>x.classList.remove("active"));

    document.querySelectorAll(".forecast-item")[i].classList.add("active");

    updateMain(d);

    applyTheme(d.type);

}

function renderForecast(){

    const c=document.getElementById("forecast");

    c.innerHTML="";

    forecastData.forEach((d,i)=>{

        const b=document.createElement("button");

        b.className="forecast-item";

        b.onclick=()=>selectDay(i);

        b.innerHTML=`

        <div>

            <div class="font-bold">${d.day}</div>

            <div class="opacity-60 text-sm">${d.date}</div>

        </div>

        <div>${themes[d.type].icon}</div>

        <div class="text-right">

            <div class="font-bold">${d.max}°</div>

            <div class="opacity-60">${d.min}°</div>

        </div>`;

        c.appendChild(b);

    });

    selectDay(0);

}

function updateMain(w){

    document.getElementById("temp").innerText=w.temp;

    document.getElementById("desc").innerText=w.desc;

    document.getElementById("feels").innerText=w.feels+"°";

    document.getElementById("humidity").innerText=w.humidity+"%";

    document.getElementById("wind").innerText=w.wind+" м/с";

    document.getElementById("pressure").innerText=w.pressure+" мм";

}

async function loadOpenMeteo(lat,lon){

    const url=`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,relative_humidity_2m,apparent_temperature,weather_code,wind_speed_10m,surface_pressure&daily=weather_code,temperature_2m_max,temperature_2m_min&timezone=auto&forecast_days=10`;

    const r=await fetch(url);

    const d=await r.json();

    const geo=await fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lon}&format=json&accept-language=ru`);

    const g=await geo.json();

    document.getElementById("city").innerText=

        g.address.city||g.address.town||g.address.state||"Неизвестно";

    forecastData=[];

    for(let i=0;i<d.daily.time.length;i++){

        let type="Clouds";

        const code=d.daily.weather_code[i];

        if(code===0) type="Clear";

        else if([61,63,65,80,81,82].includes(code)) type="Rain";

        else if([71,73,75,77,85,86].includes(code)) type="Snow";

        forecastData.push({

            type,

            day:new Date(d.daily.time[i]).toLocaleDateString("ru-RU",{weekday:"long"}),

            date:new Date(d.daily.time[i]).toLocaleDateString("ru-RU",{day:"numeric",month:"long"}),

            temp:Math.round(d.daily.temperature_2m_max[i]),

            max:Math.round(d.daily.temperature_2m_max[i]),

            min:Math.round(d.daily.temperature_2m_min[i]),

            feels:Math.round(d.current.apparent_temperature),

            humidity:d.current.relative_humidity_2m,

            wind:Math.round(d.current.wind_speed_10m),

            pressure:Math.round(d.current.surface_pressure*0.75),

            desc:themes[type].title

        });

    }

    renderForecast();

}

navigator.geolocation.getCurrentPosition(pos=>{

    coords={lat:pos.coords.latitude,lon:pos.coords.longitude};

    loadOpenMeteo(coords.lat,coords.lon);

});
